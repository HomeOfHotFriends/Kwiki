/// Load .json and .reo files from the circuits dir.
fn load_library(dir: &str) -> CircuitLibrary {
    let path = Path::new(dir);
    let mut lib = CircuitLibrary::load_dir(path).unwrap_or_else(|_| CircuitLibrary::new(dir));


    // Also load any .reo files in the dir
    if path.exists() {
        if let Ok(entries) = std::fs::read_dir(path) {
            for entry in entries.flatten() {
                let p = entry.path();
                if p.extension().and_then(|e| e.to_str()) == Some("reo") {
                    match std::fs::read_to_string(&p) {
                        Ok(src) => match parse_script(&src) {
                            Ok(circuits) => {
                                for c in circuits { lib.add(c); }
                            }
                            Err(e) => eprintln!("  ⚠ {}: {e}", p.display()),
                        },
                        Err(e) => eprintln!("  ⚠ could not read {}: {e}", p.display()),
                    }
                }
            }
        }
    }
    lib
}

fn cmd_build_demo(lib: &mut CircuitLibrary) {
    // ── NAND ──────────────────────────────────────────────────────────────
    let nand = {
        let mut b = CircuitBuilder::new("nand");
        let a   = b.input("a");
        let x   = b.input("x");
        let and = b.and(&[a, x]);
        let out = b.not(and);
        b.output("out", out);
        b.build()
    };


    // ── NOR (grown from NAND) ─────────────────────────────────────────────
    let nor = {
        let mut b = CircuitBuilder::new("nor");
        let a  = b.input("a");
        let bv = b.input("b");
        let na  = b.sub("nand", &[a, a],   1);
        let nb  = b.sub("nand", &[bv, bv], 1);
        let or  = b.sub("nand", &[na[0], nb[0]], 1);
        let out = b.sub("nand", &[or[0], or[0]], 1);
        b.output("out", out[0]);
        b.build()
    };


    // ── XOR (grown from NAND) ─────────────────────────────────────────────
    let xor = {
        let mut b = CircuitBuilder::new("xor");
        let a   = b.input("a");
        let bv  = b.input("b");
        let n   = b.sub("nand", &[a, bv], 1);
        let na  = b.sub("nand", &[a,   n[0]], 1);
        let nb  = b.sub("nand", &[bv,  n[0]], 1);
        let out = b.sub("nand", &[na[0], nb[0]], 1);
        b.output("out", out[0]);
        b.build()
    };


    for c in [&nand, &nor, &xor] {
        println!("  ✚ built '{}'", c.name);
        lib.add(c.clone());
    }
}

/// Bridge evaluation: maps a gate signal to the resulting mode (bridge_eval)
pub fn bridge_eval(gate: Gate) -> Mode {
    match gate {
        Gate::Collapse => Mode::Halt,
        Gate::Pass     => Mode::Pass,
        Gate::Flow     => Mode::Pair(PairMode::Flow),
        Gate::Deaf     => Mode::Pair(PairMode::Deaf),
        // Per core.reo: any unrecognised gate signal -> M_pair.FULL
        // All variants are exhaustive here, so no wildcard needed.
        Gate::Blind    => Mode::Pair(PairMode::Blind),
    }
}

#[derive(Debug)]
pub enum ScriptError {
    Parse { line: usize, msg: String },
    Build { line: usize, msg: String },
}


impl std::fmt::Display for ScriptError {
    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
        match self {
            Self::Parse { line, msg } => write!(f, "line {line}: parse error: {msg}"),
            Self::Build { line, msg } => write!(f, "line {line}: build error: {msg}"),
        }
    }
}


/// Parse a `.reo` script and return every `build` circuit defined in it.
///
/// Syntax:
/// ```reo
/// # comment
/// build nand {
///     in   a, x
///     let  g   = and(a, x)
///     let  out = not(g)
///     emit out
/// }
/// ```
/// Built-in gates: and  or  not  xor  nand  nor
/// Sub-circuits:   any other name is looked up as a sub-circuit (1 output)
pub fn parse(source: &str) -> Result<Vec<Circuit>, ScriptError> {
    let lines: Vec<&str> = source.lines().collect();
    let mut i = 0;
    let mut circuits = Vec::new();


    while i < lines.len() {
        let (lineno, line) = (i + 1, lines[i].trim());
        i += 1;
        if line.is_empty() || line.starts_with('#') { continue; }


        if let Some(rest) = line.strip_prefix("build ") {
            let name = rest.trim_end_matches('{').trim();
            if name.is_empty() {
                return Err(ScriptError::Parse { line: lineno, msg: "expected circuit name after 'build'".into() });
            }
            let (circuit, next_i) = parse_circuit(name, &lines, i, lineno)?;
            circuits.push(circuit);
            i = next_i;
        } else {
            return Err(ScriptError::Parse {
                line: lineno,
                msg: format!("expected 'build <name> {{', got '{line}'"),
            });
        }
    }


    Ok(circuits)
}


fn parse_circuit(
    name: &str,
    lines: &[&str],
    start: usize,
    _start_lineno: usize,
) -> Result<(Circuit, usize), ScriptError> {
    let mut b = CircuitBuilder::new(name);
    let mut wires: HashMap<String, NodeRef> = HashMap::new();
    let mut i = start;


    while i < lines.len() {
        let (lineno, line) = (i + 1, lines[i].trim());
        i += 1;
        if line.is_empty() || line.starts_with('#') { continue; }
        if line == "}" { break; }


        // ── in a, b, c ────────────────────────────────────────────────────
        if let Some(rest) = line.strip_prefix("in ").or_else(|| line.strip_prefix("in\t")) {
            for inp in split_args(rest) {
                let nr = b.input(inp);
                wires.insert(inp.to_string(), nr);
            }
            continue;
        }


        // ── let wire = gate(args) ─────────────────────────────────────────
        if let Some(rest) = line.strip_prefix("let ").or_else(|| line.strip_prefix("let\t")) {
            let (lhs, rhs) = rest.split_once('=').ok_or_else(|| ScriptError::Parse {
                line: lineno,
                msg: "expected '=' in let statement".into(),
            })?;
            let wire_name = lhs.trim();
            let (func, raw_args) = parse_call(rhs.trim(), lineno)?;
            let args = resolve_args(&raw_args, &wires, lineno)?;


            let nr = match func {
                "and"  => b.and(&args),
                "or"   => b.or(&args),
                "not"  => {
                    if args.len() != 1 {
                        return Err(ScriptError::Build { line: lineno, msg: "not() takes exactly 1 input".into() });
                    }
                    b.not(args[0])
                }
                "xor"  => b.xor(&args),
                "nand" => b.nand(&args),
                "nor"  => b.nor(&args),
                other  => {
                    // treat as sub-circuit (1 output)
                    let outs = b.sub(other, &args, 1);
                    outs[0]
                }
            };
            wires.insert(wire_name.to_string(), nr);
            continue;
        }


        // ── emit wire_name ────────────────────────────────────────────────
        if let Some(rest) = line.strip_prefix("emit ").or_else(|| line.strip_prefix("emit\t")) {
            let wire_name = rest.trim();
            let nr = wires.get(wire_name).copied().ok_or_else(|| ScriptError::Build {
                line: lineno,
                msg: format!("unknown wire '{wire_name}'"),
            })?;
            b.output(wire_name, nr);
            continue;
        }


        return Err(ScriptError::Parse {
            line: lineno,
            msg: format!("unknown statement '{line}' (expected in / let / emit / #)"),
        });
    }


    Ok((b.build(), i))
}


fn parse_call<'a>(expr: &'a str, lineno: usize) -> Result<(&'a str, Vec<&'a str>), ScriptError> {
    let (func, rest) = expr.split_once('(').ok_or_else(|| ScriptError::Parse {
        line: lineno,
        msg: format!("expected 'gate(args)', got '{expr}'"),
    })?;
    let args_str = rest.strip_suffix(')').ok_or_else(|| ScriptError::Parse {
        line: lineno,
        msg: "expected closing ')'".into(),
    })?;
    Ok((func.trim(), split_args(args_str)))
}


fn split_args(s: &str) -> Vec<&str> {
    s.split(|c| c == ',' || c == ' ' || c == '\t')
        .map(str::trim)
        .filter(|s| !s.is_empty())
        .collect()
}


fn resolve_args(
    names: &[&str],
    wires: &HashMap<String, NodeRef>,
    lineno: usize,
) -> Result<Vec<NodeRef>, ScriptError> {
    names.iter().map(|n| {
        wires.get(*n).copied().ok_or_else(|| ScriptError::Build {
            line: lineno,
            msg: format!("unknown wire '{n}'"),
        })
    }).collect()
}

use serde::{Serialize, Deserialize};


/// A signal on a reo channel — either carrying a value or suppressed.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum Signal {
    Active(BytePair),
    Suppressed(ZeroSet),
}


impl Signal {
    pub fn active(d: u8, c: u8) -> Self {
        Signal::Active(BytePair { d, c })
    }


    pub fn pos_zero() -> Self {
        Signal::Suppressed(ZeroSet::Positive)
    }


    pub fn neg_zero() -> Self {
        Signal::Suppressed(ZeroSet::Negative)
    }


    pub fn total_zero() -> Self {
        Signal::Suppressed(ZeroSet::Total)
    }


    pub fn is_active(self) -> bool {
        matches!(self, Signal::Active(_))
    }


    /// ZeroSet AND: most-suppressed state wins.
    ///
    /// Truth table (suppression hierarchy: T0 > +0/-0 > Active):
    ///   Active  & Active  = Active
    ///   Active  & +0      = +0
    ///   Active  & -0      = -0
    ///   Active  & T0      = T0
    ///   +0      & +0      = +0
    ///   +0      & -0      = T0   (conflicting polarities collapse to total)
    ///   +0      & T0      = T0
    ///   -0      & -0      = -0
    ///   -0      & T0      = T0
    ///   T0      & T0      = T0
    pub fn and(self, other: Signal) -> Signal {
        use Signal::*;
        use ZeroSet::*;
        match (self, other) {
            (Active(a), Active(b)) => Active(BytePair { d: a.d & b.d, c: a.c & b.c }),
            (Active(_), s @ Suppressed(_)) | (s @ Suppressed(_), Active(_)) => s,
            (Suppressed(Total), _) | (_, Suppressed(Total)) => Signal::total_zero(),
            (Suppressed(Positive), Suppressed(Negative))
            | (Suppressed(Negative), Suppressed(Positive)) => Signal::total_zero(),
            (s, _) => s,
        }
    }


    /// ZeroSet OR: least-suppressed state wins.
    ///
    /// Truth table:
    ///   Active  | _        = Active
    ///   +0      | -0       = Active  (opposing suppressions cancel)
    ///   +0      | +0       = +0
    ///   +0      | T0       = +0      (partial beats total in OR)
    ///   -0      | -0       = -0
    ///   -0      | T0       = -0
    ///   T0      | T0       = T0
    pub fn or(self, other: Signal) -> Signal {
        use Signal::*;
        use ZeroSet::*;
        match (self, other) {
            (Active(a), Active(b)) => Active(BytePair { d: a.d | b.d, c: a.c | b.c }),
            (Active(_), _) | (_, Active(_)) => {
                if let Active(v) = self { Active(v) } else { other }
            }
            (Suppressed(Positive), Suppressed(Negative))
            | (Suppressed(Negative), Suppressed(Positive)) => Signal::active(0, 0),
            (Suppressed(Total), s @ Suppressed(_)) | (s @ Suppressed(_), Suppressed(Total)) => s,
            (s, _) => s,
        }
    }


    /// ZeroSet NOT: inverts suppression polarity.
    ///
    ///   Active → T0    (inverting an active signal fully suppresses)
    ///   +0     → -0    (flip polarity)
    ///   -0     → +0
    ///   T0     → Active(0,0)  (inverting total suppression releases signal)
    pub fn not(self) -> Signal {
        use Signal::*;
        use ZeroSet::*;
        match self {
            Active(_)             => Signal::total_zero(),
            Suppressed(Positive)  => Signal::neg_zero(),
            Suppressed(Negative)  => Signal::pos_zero(),
            Suppressed(Total)     => Signal::active(0, 0),
        }
    }


    /// XOR: active only when exactly one input is active.
    pub fn xor(self, other: Signal) -> Signal {
        self.or(other).and(self.and(other).not())
    }
}


impl Default for Signal {
    fn default() -> Self {
        Signal::Suppressed(ZeroSet::Total)
    }
}


#[cfg(test)]
mod tests {
    use super::*;


    #[test]
    fn and_active_active() {
        let a = Signal::active(0xFF, 0xFF);
        let b = Signal::active(0x0F, 0x0F);
        assert_eq!(a.and(b), Signal::active(0x0F, 0x0F));
    }


    #[test]
    fn and_conflicting_polarity_collapses() {
        assert_eq!(Signal::pos_zero().and(Signal::neg_zero()), Signal::total_zero());
    }


    #[test]
    fn or_opposing_suppressions_cancel() {
        assert_eq!(Signal::pos_zero().or(Signal::neg_zero()), Signal::active(0, 0));
    }


    #[test]
    fn not_active_suppresses() {
        assert_eq!(Signal::active(1, 1).not(), Signal::total_zero());
    }


    #[test]
    fn not_total_zero_releases() {
        assert_eq!(Signal::total_zero().not(), Signal::active(0, 0));
    }


    #[test]
    fn not_flips_polarity() {
        assert_eq!(Signal::pos_zero().not(), Signal::neg_zero());
        assert_eq!(Signal::neg_zero().not(), Signal::pos_zero());
    }


    #[test]
    fn xor_same_inputs_suppressed() {
        let a = Signal::active(1, 1);
        assert!(a.xor(a).not().is_active()); // xor of same = suppressed, not of that = active
    }
